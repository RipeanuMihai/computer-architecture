\begin{frame}
    \frametitle{Status Flags}
    \begin{itemize}
        \item Zero flag (Z)
        \item Carry flag (C)
        \item Sign flag (S)
        \item Overflow flag (O)
        \item Parity flag (P)
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Let's do some math}
    \begin{equation}
        \begin{aligned}
            &Z=\neg | result\\
            &C=result[n+1]\\
            &S=result[n]\\
            &O=\text{operation defined}\\
            &P=\neg \oplus result
        \end{aligned}
    \end{equation}
\end{frame}

    \end{itemize}
\end{frame}
